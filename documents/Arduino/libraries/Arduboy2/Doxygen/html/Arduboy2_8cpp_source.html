232 bool Arduboy2Base::nextFrame()
233 {
234  uint8_t now = (uint8_t) millis();
235  uint8_t frameDurationMs = now - thisFrameStart;
236 
237  if (justRendered) {
238  lastFrameDurationMs = frameDurationMs;
239  justRendered = false;
240  return false;
241  }
242  else if (frameDurationMs < eachFrameMillis) {
243  // Only idle if at least a full millisecond remains, since idle() may
244  // sleep the processor until the next millisecond timer interrupt.
245  if (++frameDurationMs < eachFrameMillis) {
246  idle();
247  }
248 
249  return false;
250  }
251 
252  // pre-render
253  justRendered = true;
254  thisFrameStart = now;
255  frameCount++;
256 
257  return true;
258 }

306 void Arduboy2Base::drawPixel(int16_t x, int16_t y, uint8_t color)
307 {
308  #ifdef PIXEL_SAFE_MODE
309  if (x < 0 || x > (WIDTH-1) || y < 0 || y > (HEIGHT-1))
310  {
311  return;
312  }
313  #endif
314 
315  uint16_t row_offset;
316  uint8_t bit;
317 
318  // uint8_t row = (uint8_t)y / 8;
319  // row_offset = (row*WIDTH) + (uint8_t)x;
320  // bit = _BV((uint8_t)y % 8);
321 
322  // the above math can also be rewritten more simply as;
323  // row_offset = (y * WIDTH/8) & ~0b01111111 + (uint8_t)x;
324  // which is what the below assembler does
325 
326  // local variable for the bitshift_left array pointer,
327  // which can be declared a read-write operand
328  const uint8_t* bsl = bitshift_left;
329 
330  asm volatile
331  (
332  "mul %[width_offset], %A[y]\n"
333  "movw %[row_offset], r0\n"
334  "andi %A[row_offset], 0x80\n" // row_offset &= (~0b01111111);
335  "clr __zero_reg__\n"
336  "add %A[row_offset], %[x]\n"
337  // mask for only 0-7
338  "andi %A[y], 0x07\n"
339  // Z += y
340  "add r30, %A[y]\n"
341  "adc r31, __zero_reg__\n"
342  // load correct bitshift from program RAM
343  "lpm %[bit], Z\n"
344  : [row_offset] "=&x" (row_offset), // upper register (ANDI)
345  [bit] "=r" (bit),
346  [y] "+d" (y), // upper register (ANDI), must be writable
347  "+z" (bsl) // is modified to point to the proper shift array element
348  : [width_offset] "r" ((uint8_t)(WIDTH/8)),
349  [x] "r" ((uint8_t)x)
350  :
351  );
352 
353  if (color) {
354  sBuffer[row_offset] |= bit;
355  } else {
356  sBuffer[row_offset] &= ~ bit;
357  }
358 }

492 void Arduboy2Base::drawLine
493 (int16_t x0, int16_t y0, int16_t x1, int16_t y1, uint8_t color)
494 {
495  // bresenham's algorithm - thx wikpedia
496  bool steep = abs(y1 - y0) > abs(x1 - x0);
497  if (steep) {
498  swap(x0, y0);
499  swap(x1, y1);
500  }
501 
502  if (x0 > x1) {
503  swap(x0, x1);
504  swap(y0, y1);
505  }
506 
507  int16_t dx, dy;
508  dx = x1 - x0;
509  dy = abs(y1 - y0);
510 
511  int16_t err = dx / 2;
512  int8_t ystep;
513 
514  if (y0 < y1)
515  {
516  ystep = 1;
517  }
518  else
519  {
520  ystep = -1;
521  }
522 
523  for (; x0 <= x1; x0++)
524  {
525  if (steep)
526  {
527  drawPixel(y0, x0, color);
528  }
529  else
530  {
531  drawPixel(x0, y0, color);
532  }
533 
534  err -= dy;
535  if (err < 0)
536  {
537  y0 += ystep;
538  err += dx;
539  }
540  }

562 void Arduboy2Base::drawFastHLine
563 (int16_t x, int16_t y, uint8_t w, uint8_t color)
564 {
565  int16_t xEnd; // last x point + 1
566 
567  // Do y bounds checks
568  if (y < 0 || y >= HEIGHT)
569  return;
570 
571  xEnd = x + w;
572 
573  // Check if the entire line is not on the display
574  if (xEnd <= 0 || x >= WIDTH)
575  return;
576 
577  // Don't start before the left edge
578  if (x < 0)
579  x = 0;
580 
581  // Don't end past the right edge
582  if (xEnd > WIDTH)
583  xEnd = WIDTH;
584 
585  // calculate actual width (even if unchanged)
586  w = xEnd - x;
587 
588  // buffer pointer plus row offset + x offset
589  register uint8_t *pBuf = sBuffer + ((y / 8) * WIDTH) + x;
590 
591  // pixel mask
592  register uint8_t mask = 1 << (y & 7);
593 
594  switch (color)
595  {
596  case WHITE:
597  while (w--)
598  {
599  *pBuf++ |= mask;
600  }
601  break;
602 
603  case BLACK:
604  mask = ~mask;
605  while (w--)
606  {
607  *pBuf++ &= mask;
608  }
609  break;
610  }

713  swap(y0, y1); swap(x0, x1);
714  }
715  if (y1 > y2)
716  {
717  swap(y2, y1); swap(x2, x1);
718  }
719  if (y0 > y1)
720  {
721  swap(y0, y1); swap(x0, x1);
722  }
723 
724  if(y0 == y2)
725  { // Handle awkward all-on-same-line case as its own thing
726  a = b = x0;
727  if(x1 < a)
728  {
729  a = x1;
730  }
731  else if(x1 > b)
732  {
733  b = x1;
734  }
735  if(x2 < a)
736  {
737  a = x2;
738  }
739  else if(x2 > b)
740  {
741  b = x2;
742  }
743  drawFastHLine(a, y0, b-a+1, color);
744  return;
745  }
746 
747  int16_t dx01 = x1 - x0,
748  dy01 = y1 - y0,
749  dx02 = x2 - x0,
750  dy02 = y2 - y0,
751  dx12 = x2 - x1,
752  dy12 = y2 - y1,
753  sa = 0,
754  sb = 0;
755 
756  // For upper part of triangle, find scanline crossings for segments
757  // 0-1 and 0-2. If y1=y2 (flat-bottomed triangle), the scanline y1
758  // is included here (and second loop will be skipped, avoiding a /0
759  // error there), otherwise scanline y1 is skipped here and handled
760  // in the second loop...which also avoids a /0 error here if y0=y1
761  // (flat-topped triangle).
762  if (y1 == y2)
763  {
764  last = y1; // Include y1 scanline
765  }
766  else
767  {
768  last = y1-1; // Skip it
769  }
770 
771 
772  for(y = y0; y <= last; y++)
773  {
774  a = x0 + sa / dy01;
775  b = x0 + sb / dy02;
776  sa += dx01;
777  sb += dx02;
778 
779  if(a > b)
780  {
781  swap(a,b);
782  }
783 
784  drawFastHLine(a, y, b-a+1, color);
785  }
786 
787  // For lower part of triangle, find scanline crossings for segments
788  // 0-2 and 1-2. This loop is skipped if y1=y2.
789  sa = dx12 * (y - y1);
790  sb = dx02 * (y - y0);
791 
792  for(; y <= y2; y++)
793  {
794  a = x1 + sa / dy12;
795  b = x0 + sb / dy02;
796  sa += dx12;
797  sb += dx02;
798 
799  if(a > b)
800  {
801  swap(a,b);
802  }
803 
804  drawFastHLine(a, y, b-a+1, color);
805  }
806 }

873 struct BitStreamReader
874 {
875  const uint8_t *source;
876  uint16_t sourceIndex;
877  uint8_t bitBuffer;

886  {

889  {
890  if (this->bitBuffer == 0)

893  this->byteBuffer = pgm_read_byte(&this->source[this->sourceIndex]);
894  ++this->sourceIndex;
895  }

899

901  }

906 void Arduboy2Base::drawCompressed(int16_t sx, int16_t sy, const uint8_t *bitmap, uint8_t color)
907 {
908  // set up decompress state
909  BitStreamReader cs = BitStreamReader(bitmap);
910 
911  // read header
912  int width = (int)cs.readBits(8) + 1;
913  int height = (int)cs.readBits(8) + 1;
914  uint8_t spanColour = (uint8_t)cs.readBits(1); // starting colour
915 
916  // no need to draw at all if we're offscreen
917  if ((sx + width < 0) || (sx > WIDTH - 1) || (sy + height < 0) || (sy > HEIGHT - 1))
918  return;
919 
920  // sy = sy - (frame * height);
921  int yOffset = abs(sy) % 8;
922  int startRow = sy / 8;
923  if (sy < 0) {
924  startRow--;
925  yOffset = 8 - yOffset;
926  }
927  int rows = height / 8;
928  if ((height % 8) != 0)
929  ++rows;
930 
931  int rowOffset = 0; // +(frame*rows);
932  int columnOffset = 0;
933 
934  uint8_t byte = 0x00;
935  uint8_t bit = 0x01;
936  while (rowOffset < rows) // + (frame*rows))
937  {
938  uint16_t bitLength = 1;
939  while (cs.readBits(1) == 0)
940  bitLength += 2;
941 
942  uint16_t len = cs.readBits(bitLength) + 1; // span length
943 
944  // draw the span
945  for (uint16_t i = 0; i < len; ++i)
946  {
947  if (spanColour != 0)
948  byte |= bit;
949  bit <<= 1;
950 
951  if (bit == 0) // reached end of byte
952  {
953  // draw
954  int bRow = startRow + rowOffset;
955 
956  //if (byte) // possible optimisation
957  if ((bRow <= (HEIGHT / 8) - 1) && (bRow > -2) &&
958  (columnOffset + sx <= (WIDTH - 1)) && (columnOffset + sx >= 0))
959  {
960  int16_t offset = (bRow * WIDTH) + sx + columnOffset;
961  if (bRow >= 0)
962  {
963  int16_t index = offset;
964  uint8_t value = byte << yOffset;
965 
966  if (color != 0)
967  sBuffer[index] |= value;
968  else
969  sBuffer[index] &= ~value;
970  }
971  if ((yOffset != 0) && (bRow < (HEIGHT / 8) - 1))
972  {
973  int16_t index = offset + WIDTH;
974  uint8_t value = byte >> (8 - yOffset);
975 
976  if (color != 0)
977  sBuffer[index] |= value;
978  else
979  sBuffer[index] &= ~value;
980  }
981  }
982 
983  // iterate
984  ++columnOffset;
985  if (columnOffset >= width)
986  {
987  columnOffset = 0;
988  ++rowOffset;
989  }
990 
991  // reset byte
992  byte = 0x00;
993  bit = 0x01;
994  }
995  }
996 
997  spanColour ^= 0x01; // toggle colour bit (bit 0) for next span
998  }
999 }
