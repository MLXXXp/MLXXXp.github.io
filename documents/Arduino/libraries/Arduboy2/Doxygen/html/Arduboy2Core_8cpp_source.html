129 #ifdef ARDUBOY_SET_CPU_8MHZ
130 // If we're compiling for 8MHz we need to slow the CPU down because the
131 // hardware clock on the Arduboy is 16MHz.
132 // We also need to readjust the PLL prescaler because the Arduino USB code
133 // likely will have incorrectly set it for an 8MHz hardware clock.
134 void Arduboy2Core::setCPUSpeed8MHz()
135 {
136  uint8_t oldSREG = SREG;
137  cli(); // suspend interrupts
138  PLLCSR = _BV(PINDIV); // dissable the PLL and set prescale for 16MHz)
139  CLKPR = _BV(CLKPCE); // allow reprogramming clock
140  CLKPR = 1; // set clock divisor to 2 (0b0001)
141  PLLCSR = _BV(PLLE) | _BV(PINDIV); // enable the PLL (with 16MHz prescale)
142  SREG = oldSREG; // restore interrupts
143 }
144 #endif

146 void Arduboy2Core::bootPins()
147 {
148  uint8_t pin, mode;
149  const uint8_t *i = pinBootProgram;
150 
151  while(true) {
152  pin = pgm_read_byte(i++);
153  mode = pgm_read_byte(i++);
154  if (pin==0) break;
155  pinMode(pin, mode);
156  }
157 
158  digitalWrite(RST, HIGH);
159  delay(1); // VDD (3.3V) goes high at start, lets just chill for a ms
160  digitalWrite(RST, LOW); // bring reset low
161  delay(10); // wait 10ms
162  digitalWrite(RST, HIGH); // bring out of reset
163 }

165 void Arduboy2Core::bootOLED()
166 {
167  // setup the ports we need to talk to the OLED
168  csport = portOutputRegister(digitalPinToPort(CS));
169  cspinmask = digitalPinToBitMask(CS);

173  SPI.setClockDivider(SPI_CLOCK_DIV2);

175  LCDCommandMode();

177  // OLED to initialize it properly for Arduboy
178  for (uint8_t i = 0; i < sizeof(lcdBootProgram); i++) {
179  SPI.transfer(pgm_read_byte(lcdBootProgram + i));

249 // paint from a memory buffer, this should be FAST as it's likely what
250 // will be used by any buffer based subclass
251 void Arduboy2Core::paintScreen(uint8_t image[], bool clear)
252 {
253  uint8_t c;
254  int i = 0;
255 
256  if (clear)
257  {
258  SPDR = image[i]; // set the first SPI data byte to get things started
259  image[i++] = 0; // clear the first image byte
260  }
261  else
262  SPDR = image[i++];
263 
264  // the code to iterate the loop and get the next byte from the buffer is
265  // executed while the previous byte is being sent out by the SPI controller
266  while (i < (HEIGHT * WIDTH) / 8)
267  {
268  // get the next byte. It's put in a local variable so it can be sent as
269  // as soon as possible after the sending of the previous byte has completed
270  if (clear)
271  {
272  c = image[i];
273  // clear the byte in the image buffer
274  image[i++] = 0;
275  }
276  else
277  c = image[i++];
278 
279  while (!(SPSR & _BV(SPIF))) { } // wait for the previous byte to be sent
280 
281  // put the next byte in the SPI data register. The SPI controller will
282  // clock it out while the loop continues and gets the next byte ready
283  SPDR = c;
284  }
285  while (!(SPSR & _BV(SPIF))) { } // wait for the last byte to be sent
286 }

355 uint8_t Arduboy2Core::buttonsState()
356 {
357  uint8_t buttons;
358 
359  // using ports here is ~100 bytes smaller than digitalRead()
360 #ifdef AB_DEVKIT
361  // down, left, up
362  buttons = ((~PINB) & B01110000);
363  // right button
364  buttons = buttons | (((~PINC) & B01000000) >> 4);
365  // A and B
366  buttons = buttons | (((~PINF) & B11000000) >> 6);
367 #elif defined(ARDUBOY_10)
368  // down, up, left right
369  buttons = ((~PINF) & B11110000);
370  // A (left)
371  buttons = buttons | (((~PINE) & B01000000) >> 3);
372  // B (right)
373  buttons = buttons | (((~PINB) & B00010000) >> 2);
374 #endif
375
