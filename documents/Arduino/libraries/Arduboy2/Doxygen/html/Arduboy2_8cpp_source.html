250 bool Arduboy2Base::nextFrame()
251 {
252  uint8_t now = (uint8_t) millis();
253  uint8_t frameDurationMs = now - thisFrameStart;
254 
255  if (justRendered) {
256  lastFrameDurationMs = frameDurationMs;
257  justRendered = false;
258  return false;
259  }
260  else if (frameDurationMs < eachFrameMillis) {
261  // Only idle if at least a full millisecond remains, since idle() may
262  // sleep the processor until the next millisecond timer interrupt.
263  if (++frameDurationMs < eachFrameMillis) {
264  idle();
265  }
266 
267  return false;
268  }
269 
270  // pre-render
271  justRendered = true;
272  thisFrameStart = now;
273  frameCount++;
274 
275  return true;
276 }

521 void Arduboy2Base::drawLine
522 (int16_t x0, int16_t y0, int16_t x1, int16_t y1, uint8_t color)
523 {
524  // bresenham's algorithm - thx wikpedia
525  bool steep = abs(y1 - y0) > abs(x1 - x0);
526  if (steep) {
527  swap(x0, y0);
528  swap(x1, y1);
529  }
530 
531  if (x0 > x1) {
532  swap(x0, x1);
533  swap(y0, y1);
534  }
535 
536  int16_t dx, dy;
537  dx = x1 - x0;
538  dy = abs(y1 - y0);
539 
540  int16_t err = dx / 2;
541  int8_t ystep;
542 
543  if (y0 < y1)
544  {
545  ystep = 1;
546  }
547  else
548  {
549  ystep = -1;
550  }
551 
552  for (; x0 <= x1; x0++)
553  {
554  if (steep)
555  {
556  drawPixel(y0, x0, color);
557  }
558  else
559  {
560  drawPixel(x0, y0, color);
561  }
562

564  if (err < 0)

591 void Arduboy2Base::drawFastHLine
592 (int16_t x, int16_t y, uint8_t w, uint8_t color)
593 {
594  int16_t xEnd; // last x point + 1
595 
596  // Do y bounds checks
597  if (y < 0 || y >= HEIGHT)
598  return;
599 
600  xEnd = x + w;
601 
602  // Check if the entire line is not on the display
603  if (xEnd <= 0 || x >= WIDTH)
604  return;
605 
606  // Don't start before the left edge
607  if (x < 0)
608  x = 0;
609 
610  // Don't end past the right edge
611  if (xEnd > WIDTH)
612  xEnd = WIDTH;
613 
614  // calculate actual width (even if unchanged)
615  w = xEnd - x;
616 
617  // buffer pointer plus row offset + x offset
618  register uint8_t *pBuf = sBuffer + ((y / 8) * WIDTH) + x;
619 
620  // pixel mask
621  register uint8_t mask = 1 << (y & 7);
622 
623  switch (color)
624  {
625  case WHITE:
626  while (w--)
627  {
628  *pBuf++ |= mask;
629  }
630  break;
631 
632  case BLACK:
633  mask = ~mask;
634  while (w--)
635  {
636  *pBuf++ &= mask;
637  }
638  break;
639  }
640 }

734 void Arduboy2Base::fillTriangle
735 (int16_t x0, int16_t y0, int16_t x1, int16_t y1, int16_t x2, int16_t y2, uint8_t color)
736 {
737 
738  int16_t a, b, y, last;
739  // Sort coordinates by Y order (y2 >= y1 >= y0)
740  if (y0 > y1)
741  {
742  swap(y0, y1); swap(x0, x1);
743  }
744  if (y1 > y2)
745  {
746  swap(y2, y1); swap(x2, x1);
747  }
748  if (y0 > y1)
749  {
750  swap(y0, y1); swap(x0, x1);
751  }
752 
753  if(y0 == y2)
754  { // Handle awkward all-on-same-line case as its own thing
755  a = b = x0;
756  if(x1 < a)
757  {
758  a = x1;
759  }
760  else if(x1 > b)
761  {
762  b = x1;
763  }
764  if(x2 < a)
765  {
766  a = x2;
767  }
768  else if(x2 > b)
769  {
770  b = x2;
771  }
772  drawFastHLine(a, y0, b-a+1, color);
773  return;
774  }
775 
776  int16_t dx01 = x1 - x0,
777  dy01 = y1 - y0,
778  dx02 = x2 - x0,
779  dy02 = y2 - y0,
780  dx12 = x2 - x1,
781  dy12 = y2 - y1,
782  sa = 0,
783  sb = 0;
784 
785  // For upper part of triangle, find scanline crossings for segments
786  // 0-1 and 0-2. If y1=y2 (flat-bottomed triangle), the scanline y1
787  // is included here (and second loop will be skipped, avoiding a /0
788  // error there), otherwise scanline y1 is skipped here and handled
789  // in the second loop...which also avoids a /0 error here if y0=y1
790  // (flat-topped triangle).
791  if (y1 == y2)
792  {
793  last = y1; // Include y1 scanline
794  }
795  else
796  {
797  last = y1-1; // Skip it
798  }
799 
800 
801  for(y = y0; y <= last; y++)
802  {
803  a = x0 + sa / dy01;
804  b = x0 + sb / dy02;
805  sa += dx01;
806  sb += dx02;
807 
808  if(a > b)
809  {
810  swap(a,b);
811  }
812 
813  drawFastHLine(a, y, b-a+1, color);
814  }
815 
816  // For lower part of triangle, find scanline crossings for segments
817  // 0-2 and 1-2. This loop is skipped if y1=y2.
818  sa = dx12 * (y - y1);
819  sb = dx02 * (y - y0);
820 
821  for(; y <= y2; y++)
822  {
823  a = x1 + sa / dy12;
824  b = x0 + sb / dy02;
825  sa += dx12;
826  sb += dx02;
827 
828  if(a > b)
829  {
830  swap(a,b);
831  }
832 
833  drawFastHLine(a, y, b-a+1, color);
834  }
835 }

905  uint16_t sourceIndex;
906  uint8_t bitBuffer;

915  {

923  ++this->sourceIndex;

935 void Arduboy2Base::drawCompressed(int16_t sx, int16_t sy, const uint8_t *bitmap, uint8_t color)
936 {
937  // set up decompress state
938  BitStreamReader cs = BitStreamReader(bitmap);
939 
940  // read header
941  int width = (int)cs.readBits(8) + 1;
942  int height = (int)cs.readBits(8) + 1;
943  uint8_t spanColour = (uint8_t)cs.readBits(1); // starting colour
944 
945  // no need to draw at all if we're offscreen
946  if ((sx + width < 0) || (sx > WIDTH - 1) || (sy + height < 0) || (sy > HEIGHT - 1))
947  return;
948 
949  // sy = sy - (frame * height);
950  int yOffset = abs(sy) % 8;
951  int startRow = sy / 8;
952  if (sy < 0) {
953  startRow--;
954  yOffset = 8 - yOffset;
955  }
956  int rows = height / 8;
957  if ((height % 8) != 0)
958  ++rows;
959 
960  int rowOffset = 0; // +(frame*rows);
961  int columnOffset = 0;
962 
963  uint8_t byte = 0x00;
964  uint8_t bit = 0x01;
965  while (rowOffset < rows) // + (frame*rows))
966  {
967  uint16_t bitLength = 1;
968  while (cs.readBits(1) == 0)
969  bitLength += 2;
970 
971  uint16_t len = cs.readBits(bitLength) + 1; // span length
972 
973  // draw the span
974  for (uint16_t i = 0; i < len; ++i)
975  {
976  if (spanColour != 0)
977  byte |= bit;
978  bit <<= 1;
979 
980  if (bit == 0) // reached end of byte
981  {
982  // draw
983  int bRow = startRow + rowOffset;
984 
985  //if (byte) // possible optimisation
986  if ((bRow <= (HEIGHT / 8) - 1) && (bRow > -2) &&
987  (columnOffset + sx <= (WIDTH - 1)) && (columnOffset + sx >= 0))
988  {
989  int16_t offset = (bRow * WIDTH) + sx + columnOffset;
990  if (bRow >= 0)
991  {
992  int16_t index = offset;
993  uint8_t value = byte << yOffset;
994 
995  if (color != 0)
996  sBuffer[index] |= value;
997  else
998  sBuffer[index] &= ~value;
999  }
1000  if ((yOffset != 0) && (bRow < (HEIGHT / 8) - 1))
1001  {
1002  int16_t index = offset + WIDTH;
1003  uint8_t value = byte >> (8 - yOffset);
1004 
1005  if (color != 0)
1006  sBuffer[index] |= value;
1007  else
1008  sBuffer[index] &= ~value;
1009  }
1010  }
1011 
1012  // iterate
1013  ++columnOffset;
1014  if (columnOffset >= width)
1015  {
1016  columnOffset = 0;
1017  ++rowOffset;
1018  }
1019 
1020  // reset byte
1021  byte = 0x00;
1022  bit = 0x01;
1023  }
1024  }
1025 
1026  spanColour ^= 0x01; // toggle colour bit (bit 0) for next span
1027  }
1028 }
