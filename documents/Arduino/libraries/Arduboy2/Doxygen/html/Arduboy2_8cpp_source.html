145 bool Arduboy2Base::nextFrame()
146 {
147  unsigned long now = millis();
148 
149  // post render
150  if (post_render) {
151  lastFrameDurationMs = now - lastFrameStart;
152  frameCount++;
153  post_render = false;
154  }
155 
156  // if it's not time for the next frame yet
157  if (now < nextFrameStart) {
158  // if we have more than 1ms to spare, lets sleep
159  // we should be woken up by timer0 every 1ms, so this should be ok
160  if ((uint8_t)(nextFrameStart - now) > 1)
161  idle();
162  return false;
163  }
164 
165  // pre-render
166  nextFrameStart = now + eachFrameMillis;
167  lastFrameStart = now;
168  post_render = true;
169  return post_render;
170 }

370 void Arduboy2Base::drawLine
371 (int16_t x0, int16_t y0, int16_t x1, int16_t y1, uint8_t color)
372 {
373  // bresenham's algorithm - thx wikpedia
374  bool steep = abs(y1 - y0) > abs(x1 - x0);
375  if (steep) {
376  swap(x0, y0);
377  swap(x1, y1);
378  }
379 
380  if (x0 > x1) {
381  swap(x0, x1);
382  swap(y0, y1);
383  }
384 
385  int16_t dx, dy;
386  dx = x1 - x0;
387  dy = abs(y1 - y0);
388 
389  int16_t err = dx / 2;
390  int8_t ystep;
391 
392  if (y0 < y1)
393  {
394  ystep = 1;
395  }
396  else
397  {
398  ystep = -1;
399  }
400 
401  for (; x0 <= x1; x0++)
402  {
403  if (steep)
404  {
405  drawPixel(y0, x0, color);
406  }
407  else
408  {
409  drawPixel(x0, y0, color);
410  }
411 
412  err -= dy;
413  if (err < 0)
414  {
415  y0 += ystep;
416  err += dx;
417  }
418  }

440 void Arduboy2Base::drawFastHLine
441 (int16_t x, int16_t y, uint8_t w, uint8_t color)
442 {
443  int16_t xEnd; // last x point + 1
444 
445  // Do y bounds checks
446  if (y < 0 || y >= HEIGHT)
447  return;
448 
449  xEnd = x + w;
450 
451  // Check if the entire line is not on the display
452  if (xEnd <= 0 || x >= WIDTH)
453  return;
454 
455  // Don't start before the left edge
456  if (x < 0)
457  x = 0;
458 
459  // Don't end past the right edge
460  if (xEnd > WIDTH)
461  xEnd = WIDTH;
462 
463  // calculate actual width (even if unchanged)
464  w = xEnd - x;
465 
466  // buffer pointer plus row offset + x offset
467  register uint8_t *pBuf = sBuffer + ((y / 8) * WIDTH) + x;
468 
469  // pixel mask
470  register uint8_t mask = 1 << (y & 7);
471 
472  switch (color)
473  {
474  case WHITE:
475  while (w--)
476  {
477  *pBuf++ |= mask;
478  }
479  break;
480 
481  case BLACK:
482  mask = ~mask;
483  while (w--)
484  {
485  *pBuf++ &= mask;
486  }
487  break;
488  }

573 void Arduboy2Base::fillTriangle
574 (int16_t x0, int16_t y0, int16_t x1, int16_t y1, int16_t x2, int16_t y2, uint8_t color)
575 {
576 
577  int16_t a, b, y, last;
578  // Sort coordinates by Y order (y2 >= y1 >= y0)
579  if (y0 > y1)
580  {
581  swap(y0, y1); swap(x0, x1);
582  }
583  if (y1 > y2)
584  {
585  swap(y2, y1); swap(x2, x1);
586  }
587  if (y0 > y1)
588  {
589  swap(y0, y1); swap(x0, x1);
590  }
591 
592  if(y0 == y2)
593  { // Handle awkward all-on-same-line case as its own thing
594  a = b = x0;
595  if(x1 < a)
596  {
597  a = x1;
598  }
599  else if(x1 > b)
600  {
601  b = x1;
602  }
603  if(x2 < a)
604  {
605  a = x2;
606  }
607  else if(x2 > b)
608  {
609  b = x2;
610  }
611  drawFastHLine(a, y0, b-a+1, color);
612  return;
613  }
614 
615  int16_t dx01 = x1 - x0,
616  dy01 = y1 - y0,
617  dx02 = x2 - x0,
618  dy02 = y2 - y0,
619  dx12 = x2 - x1,
620  dy12 = y2 - y1,
621  sa = 0,
622  sb = 0;
623 
624  // For upper part of triangle, find scanline crossings for segments
625  // 0-1 and 0-2. If y1=y2 (flat-bottomed triangle), the scanline y1
626  // is included here (and second loop will be skipped, avoiding a /0
627  // error there), otherwise scanline y1 is skipped here and handled
628  // in the second loop...which also avoids a /0 error here if y0=y1
629  // (flat-topped triangle).
630  if (y1 == y2)
631  {
632  last = y1; // Include y1 scanline
633  }
634  else
635  {
636  last = y1-1; // Skip it
637  }
638 
639 
640  for(y = y0; y <= last; y++)
641  {
642  a = x0 + sa / dy01;
643  b = x0 + sb / dy02;
644  sa += dx01;
645  sb += dx02;
646 
647  if(a > b)
648  {
649  swap(a,b);
650  }
651 
652  drawFastHLine(a, y, b-a+1, color);
653  }
654 
655  // For lower part of triangle, find scanline crossings for segments
656  // 0-2 and 1-2. This loop is skipped if y1=y2.
657  sa = dx12 * (y - y1);
658  sb = dx02 * (y - y0);
659 
660  for(; y <= y2; y++)
661  {
662  a = x1 + sa / dy12;
663  b = x0 + sb / dy02;
664  sa += dx12;
665  sb += dx02;
666

670  }

676 void Arduboy2Base::drawBitmap
677 (int16_t x, int16_t y, const uint8_t *bitmap, uint8_t w, uint8_t h,
678  uint8_t color)
679 {

681  if (x+w < 0 || x > WIDTH-1 || y+h < 0 || y > HEIGHT-1)
682  return;

685  int sRow = y / 8;
686  if (y < 0) {
687  sRow--;
688  yOffset = 8 - yOffset;
689  }

691  if (h%8!=0) rows++;
692  for (int a = 0; a < rows; a++) {
693  int bRow = sRow + a;
694  if (bRow > (HEIGHT/8)-1) break;

702  else if (color == BLACK)
703  sBuffer[(bRow*WIDTH) + x + iCol] &= ~(pgm_read_byte(bitmap+(a*w)+iCol) << yOffset);
704  else
705  sBuffer[(bRow*WIDTH) + x + iCol] ^= pgm_read_byte(bitmap+(a*w)+iCol) << yOffset;
706  }

708  if (color == WHITE)
709  sBuffer[((bRow+1)*WIDTH) + x + iCol] |= pgm_read_byte(bitmap+(a*w)+iCol) >> (8-yOffset);
710  else if (color == BLACK)

752  {

766 void Arduboy2Base::drawCompressed(int16_t sx, int16_t sy, const uint8_t *bitmap, uint8_t color)
767 {
768  int bl, len;
769  int col;
770  int i;
771  int a, iCol;
772  int byte = 0;
773  int bit = 0;
774  int w, h;
775 
776  // set up decompress state
777 
778  cs.src = bitmap;
779  cs.bit = 0x100;
780  cs.byte = 0;
781  cs.src_pos = 0;
782 
783  // read header
784 
785  w = getval(8) + 1;
786  h = getval(8) + 1;
787 
788  col = getval(1); // starting colour
789 
790  // no need to draw at all if we're offscreen
791  if (sx + w < 0 || sx > WIDTH - 1 || sy + h < 0 || sy > HEIGHT - 1)
792  return;
793 
794  // sy = sy - (frame*h);
795 
796  int yOffset = abs(sy) % 8;
797  int sRow = sy / 8;
798  if (sy < 0) {
799  sRow--;
800  yOffset = 8 - yOffset;
801  }
802  int rows = h / 8;
803  if (h % 8 != 0) rows++;
804 
805  a = 0; // +(frame*rows);
806  iCol = 0;
807 
808  byte = 0; bit = 1;
809  while (a < rows) // + (frame*rows))
810  {
811  bl = 1;
812  while (!getval(1))
813  bl += 2;
814 
815  len = getval(bl) + 1; // span length
816 
817  // draw the span
818 
819 
820  for (i = 0; i < len; i++)
821  {
822  if (col)
823  byte |= bit;
824  bit <<= 1;
825 
826  if (bit == 0x100) // reached end of byte
827  {
828  // draw
829 
830  int bRow = sRow + a;
831 
832  //if (byte) // possible optimisation
833  if (bRow <= (HEIGHT / 8) - 1)
834  if (bRow > -2)
835  if (iCol + sx <= (WIDTH - 1))
836  if (iCol + sx >= 0) {
837 
838  if (bRow >= 0)
839  {
840  if (color)
841  sBuffer[(bRow * WIDTH) + sx + iCol] |= byte << yOffset;
842  else
843  sBuffer[(bRow * WIDTH) + sx + iCol] &= ~(byte << yOffset);
844  }
845  if (yOffset && bRow < (HEIGHT / 8) - 1 && bRow > -2)
846  {
847  if (color)
848  sBuffer[((bRow + 1)*WIDTH) + sx + iCol] |= byte >> (8 - yOffset);
849  else
850  sBuffer[((bRow + 1)*WIDTH) + sx + iCol] &= ~(byte >> (8 - yOffset));
851  }
852  }
853 
854  // iterate
855  iCol ++;
856  if (iCol >= w)
857  {
858  iCol = 0;
859  a ++;
860  }
861 
862  // reset byte
863  byte = 0; bit = 1;
864  }
865  }
866 
867  col = 1 - col; // toggle colour for next span
868  }
869 }
