98 {
99  uint8_t oldSREG = SREG;

194 void Arduboy2Core::bootOLED()
195 {
196  // reset the display
197  delayShort(5); // reset pin should be low here. let it stay low a while
198  bitSet(RST_PORT, RST_BIT); // set high to come out of reset
199  delayShort(5); // wait a while
200 
201  // select the display (permanently, since nothing else is using SPI)
202  bitClear(CS_PORT, CS_BIT);
203 
204  // run our customized boot-up command sequence against the
205  // OLED to initialize it properly for Arduboy
206  LCDCommandMode();
207  for (uint8_t i = 0; i < sizeof(lcdBootProgram); i++) {
208  SPItransfer(pgm_read_byte(lcdBootProgram + i));
209  }
210  LCDDataMode();
211 }

232 void Arduboy2Core::SPItransfer(uint8_t data)
233 {
234  SPDR = data;
235  /*
236  * The following NOP introduces a small delay that can prevent the wait
237  * loop form iterating when running at the maximum speed. This gives
238  * about 10% more speed, even if it seems counter-intuitive. At lower
239  * speeds it is unnoticed.
240  */
241  asm volatile("nop");
242  while (!(SPSR & _BV(SPIF))) { } // wait
243 }

314 // paint from a memory buffer, this should be FAST as it's likely what
315 // will be used by any buffer based subclass
316 void Arduboy2Core::paintScreen(uint8_t image[], bool clear)
317 {
318  uint8_t c;
319  int i = 0;
320 
321  if (clear)
322  {
323  SPDR = image[i]; // set the first SPI data byte to get things started
324  image[i++] = 0; // clear the first image byte
325  }
326  else
327  SPDR = image[i++];
328 
329  // the code to iterate the loop and get the next byte from the buffer is
330  // executed while the previous byte is being sent out by the SPI controller
331  while (i < (HEIGHT * WIDTH) / 8)
332  {
333  // get the next byte. It's put in a local variable so it can be sent as
334  // as soon as possible after the sending of the previous byte has completed
335  if (clear)
336  {
337  c = image[i];
338  // clear the byte in the image buffer
339  image[i++] = 0;
340  }
341  else
342  c = image[i++];
343 
344  while (!(SPSR & _BV(SPIF))) { } // wait for the previous byte to be sent
345 
346  // put the next byte in the SPI data register. The SPI controller will
347  // clock it out while the loop continues and gets the next byte ready
348  SPDR = c;
349  }
350  while (!(SPSR & _BV(SPIF))) { } // wait for the last byte to be sent
351 }

449 uint8_t Arduboy2Core::buttonsState()
450 {
451  uint8_t buttons;
452 
453  // using ports here is ~100 bytes smaller than digitalRead()
454 #ifdef AB_DEVKIT
455  // down, left, up
456  buttons = ((~PINB) & B01110000);
457  // right button
458  buttons = buttons | (((~PINC) & B01000000) >> 4);
459  // A and B
460  buttons = buttons | (((~PINF) & B11000000) >> 6);
461 #elif defined(ARDUBOY_10)
462  // down, up, left right
463  buttons = ((~PINF) & B11110000);
464  // A (left)
465  buttons = buttons | (((~PINE) & B01000000) >> 3);
466  // B (right)
467  buttons = buttons | (((~PINB) & B00010000) >> 2);
468 #endif
469 
470  return buttons;
471 }
